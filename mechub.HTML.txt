  <script>
        let currentUser = null;
        let currentRole = 'driver';
        let allData = [];

        // Data SDK Handler
        const dataHandler = {
            onDataChanged(data) {
                allData = data;
                updateUI();
                updateDataDisplay();
            }
        };

        // Initialize
        async function init() {
            const result = await window.dataSdk.init(dataHandler);
            if (!result.isOk) {
                console.error('SDK init failed');
                showError('Failed to initialize', 'driverError');
            }
            lucide.createIcons();
        }

        // Switch between driver/mechanic
        function switchRole(role) {
            currentRole = role;
            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');
            
            document.getElementById('driverForm').style.display = role === 'driver' ? 'block' : 'none';
            document.getElementById('mechanicForm').style.display = role === 'mechanic' ? 'block' : 'none';
        }

        // Register Driver
        async function registerDriver() {
            const name = document.getElementById('driverName').value;
            const email = document.getElementById('driverEmail').value;
            const phone = document.getElementById('driverPhone').value;
            const location = document.getElementById('driverLocation').value;
            const car = document.getElementById('driverCar').value;

            if (!name || !email || !phone || !location || !car) {
                showError('All fields required!', 'driverError');
                return;
            }

            const result = await window.dataSdk.create({
                type: 'driver',
                name, email, phone, location, car,
                createdAt: new Date().toISOString(),
                rating: 5,
                bookings: '0'
            });

            if (result.isOk) {
                currentUser = { type: 'driver', name };
                document.getElementById('userDisplay').textContent = `🚗 ${name}`;
                showSuccess('Registered!', 'driverSuccess');
                document.getElementById('driverForm').reset?.();
                setTimeout(() => updateUI(), 500);
            } else {
                showError(result.error.message, 'driverError');
            }
        }

        // Register Mechanic
        async function registerMechanic() {
            const name = document.getElementById('mechanicName').value;
            const email = document.getElementById('mechanicEmail').value;
            const phone = document.getElementById('mechanicPhone').value;
            const location = document.getElementById('mechanicLocation').value;
            const service = document.getElementById('mechanicService').value;
            const rate = document.getElementById('mechanicRate').value;

            if (!name || !email || !phone || !location || !service || !rate) {
                showError('All fields required!', 'mechanicError');
                return;
            }

            const result = await window.dataSdk.create({
                type: 'mechanic',
                name, email, phone, location, service,
                rate: rate.toString(),
                createdAt: new Date().toISOString(),
                rating: 5,
                jobs: '0'
            });

            if (result.isOk) {
                currentUser = { type: 'mechanic', name };
                document.getElementById('userDisplay').textContent = `🔧 ${name}`;
                showSuccess('Registered!', 'mechanicSuccess');
                setTimeout(() => updateUI(), 500);
            } else {
                showError(result.error.message, 'mechanicError');
            }
        }

        // Search Mechanics
        function searchMechanics() {
            const location = document.getElementById('searchLocation').value;
            const service = document.getElementById('serviceType').value;

            if (!location) {
                showError('Enter location!', 'driverError');
                return;
            }

            const mechanics = allData.filter(d => 
                d.type === 'mechanic' && 
                d.location?.toLowerCase().includes(location.toLowerCase())
            );

            const list = document.getElementById('mechanicsList');
            if (mechanics.length === 0) {
                list.innerHTML = '<p class="text-gray-400">No mechanics found</p>';
            } else {
                list.innerHTML = mechanics.map(m => `
                    <div class="bg-blue-900 p-3 rounded border border-blue-700">
                        <div class="flex justify-between">
                            <div>
                                <p class="font-bold">${m.name}</p>
                                <p class="text-sm text-gray-300">${m.service} - ₹${m.rate}/hr</p>
                                <p class="text-xs text-yellow-400">⭐ ${m.rating}</p>
                            </div>
                            <button class="btn-primary text-sm" onclick="bookMechanic('${m.__backendId}')">Book</button>
                        </div>
                    </div>
                `).join('');
            }

            document.getElementById('resultsSection').style.display = 'block';
        }

        // Book Mechanic
        async function bookMechanic(mechanicId) {
            if (!currentUser || currentUser.type !== 'driver') {
                showError('Register as driver first!', 'driverError');
                return;
            }

            const mechanic = allData.find(d => d.__backendId === mechanicId);
            const result = await window.dataSdk.create({
                type: 'booking',
                driverName: currentUser.name,
                mechanicName: mechanic.name,
                mechanicId: mechanicId,
                status: 'pending',
                rate: mechanic.rate,
                createdAt: new Date().toISOString(),
                completedAt: ''
            });

            if (result.isOk) {
                showSuccess(`Booked ${mechanic.name}!`, 'driverSuccess');
            }
        }

        // Complete Booking
        async function completeBooking(bookingId) {
            const booking = allData.find(b => b.__backendId === bookingId);
            const updatedBooking = {
                ...booking,
                status: 'completed',
                completedAt: new Date().toISOString()
            };

            const result = await window.dataSdk.update(updatedBooking);
            if (result.isOk) {
                showSuccess('Booking completed!', 'driverSuccess');
            }
        }

        // Rate Service
        async function rateService(bookingId) {
            const rating = prompt('Rate 1-5 stars:');
            if (!rating || rating < 1 || rating > 5) return;

            const booking = allData.find(b => b.__backendId === bookingId);
            const updatedBooking = {
                ...booking,
                rating: rating.toString()
            };

            const result = await window.dataSdk.update(updatedBooking);
            if (result.isOk) {
                showSuccess(`Rated ${rating} stars!`, 'driverSuccess');
            }
        }

        // Update UI
        function updateUI() {
            if (!currentUser) {
                document.getElementById('authSection').style.display = 'block';
                document.getElementById('searchSection').style.display = 'none';
                document.getElementById('bookingsSection').style.display = 'none';
                document.getElementById('mechanicDashboard').style.display = 'none';
            } else if (currentUser.type === 'driver') {
                document.getElementById('authSection').style.display = 'none';
                document.getElementById('searchSection').style.display = 'block';
                document.getElementById('bookingsSection').style.display = 'block';
                
                const bookings = allData.filter(d => d.type === 'booking' && d.driverName === currentUser.name);
                document.getElementById('bookingsList').innerHTML = bookings.length === 0 
                    ? '<p class="text-gray-400">No bookings yet</p>'
                    : bookings.map(b => `
                        <div class="bg-green-900 p-3 rounded border border-green-700">
                            <p class="font-bold">${b.mechanicName}</p>
                            <p class="text-sm text-gray-300">Status: ${b.status}</p>
                            <p class="text-sm">₹${b.rate}/hr</p>
                            ${b.status === 'pending' ? `<button class="btn-secondary text-sm mt-2" onclick="completeBooking('${b.__backendId}')">Complete</button>` : ''}
                            ${b.status === 'completed' && !b.rating ? `<button class="btn-secondary text-sm mt-2" onclick="rateService('${b.__backendId}')">Rate</button>` : ''}
                        </div>
                    `).join('');
            } else if (currentUser.type === 'mechanic') {
                document.getElementById('authSection').style.display = 'none';
                document.getElementById('searchSection').style.display = 'none';
                document.getElementById('mechanicDashboard').style.display = 'block';
                
                const jobs = allData.filter(d => d.type === 'booking' && d.mechanicName === currentUser.name);
                document.getElementById('jobsList').innerHTML = jobs.length === 0
                    ? '<p class="text-gray-400">No jobs yet</p>'
                    : jobs.map(j => `
                        <div class="bg-purple-900 p-3 rounded border border-purple-700">
                            <p class="font-bold">${j.driverName}</p>
                            <p class="text-sm text-gray-300">Status: ${j.status}</p>
                            <p class="text-sm">₹${j.rate}/hr</p>
                            ${j.rating ? `<p class="text-yellow-400">⭐ ${j.rating} stars</p>` : ''}
                        </div>
                    `).join('');
            }
        }

        // Show Success
        function showSuccess(msg, elementId) {
            const el = document.getElementById(elementId);
            if (el) {
                el.textContent = msg;
                el.style.display = 'block';
                setTimeout(() => el.style.display = 'none', 3000);
            }
        }

        // Show Error
        function showError(msg, elementId) {
            const el = document.getElementById(elementId);
            if (el) {
                el.textContent = '❌ ' + msg;
                el.style.display = 'block';
                setTimeout(() => el.style.display = 'none', 3000);
            }
        }

        // Update Data Display
        function updateDataDisplay() {
            document.getElementById('dataContent').textContent = JSON.stringify(allData, null, 2);
        }

        // Start
        init();
    </script>
